	add = {
		desc = "BUILDING_LEVEL_BASE"
		value = 1
	}
	add = {
		desc = "BUILDING_LEVEL_DEVELOPMENT"
		value = development
		multiply = 0.1
	}
	if = {
		# Open vegetation bonus
		limit = {
			OR = {
				vegetation = grasslands
				vegetation = farmland
				vegetation = sparse
			}
		}
		add = {
			desc = "BUILDING_LEVEL_RGO_MAX"
			value = max_rgo_workers
			multiply = 0.35
		}
	}
	else = {
		add = {
			desc = "BUILDING_LEVEL_RGO_MAX"
			value = max_rgo_workers
			multiply = 0.2
		}
	}
	if = {
		limit = {
			OR = {
				climate = mediterranean
				climate = oceanic

				climate = subtropical
			}
		}
		multiply = {
			desc = "BUILDING_LEVEL_FAVORABLE_CLIMATE"
			value = 1.4
		}
	}
	if = {
		limit = {
			OR = {
				has_river = yes
				is_adjacent_to_lake = yes
			}
		}
		multiply = {
			desc = "BUILDING_LEVEL_HAS_IRRIGATION"
			value = 1.4
		}
	}
	add = {
		any_building_in_location = {
			building_category = farming_building_category
			add = 1
		}
		multiply = -1
	}
