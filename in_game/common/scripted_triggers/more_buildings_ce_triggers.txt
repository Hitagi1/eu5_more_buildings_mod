mb_potato_trigger = {
	NOT = {
		raw_material ?= goods:potato
	}
	NOT = {
		vegetation = desert
	}
	area = {
		is_area_sea = no
	}
	NOR = {
		climate = tropical
		climate = arid
		AND = {
			climate = cold_arid
			OR = {
				has_river = no
				is_adjacent_to_lake = no
			}
		}
		climate = arctic
	}
}
