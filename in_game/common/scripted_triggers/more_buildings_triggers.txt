old_world_crops_building_trigger = {
	OR = {
		NOT = {
			culture = {
				is_culture_native_american = yes
			}
		}
		has_variable = is_in_columbian_exchange
		current_age_or_later = {
			age = age_6_revolutions
		}
	}
}

new_world_crops_building_trigger = {
	OR = {
		culture = {
			is_culture_native_american = yes
		}
		has_variable = is_in_columbian_exchange
		current_age_or_later = {
			age = age_6_revolutions
		}
	}
}

market_has_good = {
	market ?= {
		is_produced_in_market = goods:$good$
		is_traded_in_market = goods:$good$
	}
}
